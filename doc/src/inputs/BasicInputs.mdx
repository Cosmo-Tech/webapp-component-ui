# Basic Inputs

This part will list all basic inputs that you can use.

## BasicEnumInput

### Props

<Props of={BasicEnumInput} />

### Basic Usage

<Playground>
  {() => {
    class BasicEnumInputExample extends React.Component {
      constructor(props) {
        super(props);
        this.state = { value: 'USD' };
      }
      changeEnumValue(value) {
        this.setState({ value: value });
      }
      render() {
        return (
          <>
            <BasicEnumInput
              label="Here is a label"
              value={this.state.value}
              tooltipText={'Basic enum input tooltip'}
              changeEnumField={(value) => this.changeEnumValue(value)}
              enumValues={[
                {
                  key: 'USD',
                  value: '$',
                },
                {
                  key: 'EUR',
                  value: '€',
                },
                {
                  key: 'BTC',
                  value: '฿',
                },
                {
                  key: 'YUAN',
                  value: '¥',
                },
              ]}
            />
          </>
        );
      }
    }
    return <BasicEnumInputExample />;
  }}
</Playground>
