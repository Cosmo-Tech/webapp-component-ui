# Basic Inputs

This part will list all basic inputs that you can use.

## BasicEnumInput

### Props

<Props of={BasicEnumInput} />

### Basic Usage

<Playground>
  {() => {
    class BasicEnumInputExample extends React.Component {
      constructor(props) {
        super(props);
        this.state = { value: '' };
      }
      changeEnumValue(value) {
        this.setState({ value: value });
      }
      render() {
        return (
          <>
            <BasicEnumInput
              label="Here is a label"
              value={this.state.value}
              changeEnumField={(value) => this.changeEnumValue(value)}
              textFieldProps={{
                id: 'basic-text-input-id',
              }}
              enumValues={[
                {
                  key: 'USD',
                  value: 'A',
                },
                {
                  key: 'EUR',
                  value: 'Z',
                },
              ]}
            />
          </>
        );
      }
    }
    return <BasicEnumInputExample />;
  }}
</Playground>

    class BasicDateInputExample extends React.Component {
      constructor(props) {
        super(props);
        this.state = { value: '09/03/2021' };

      changeDateValue(value) {

      }
      render() {
        return (
          <>
            <BasicDateInput
              value={this.state.value}
              changeSelectedDate={(value) => this.changeDateValue(value)}
              format={'MM/dd/yyyy'}
              label={'Pick a date'}
              dateProps={{
                id: 'basic-date-input-id',
                minDate: new Date('2014-01-01'),
                maxDate: new Date('2023-01-01'),
                minDateMessage: 'Minimum date is not respected',
                maxDateMessage: 'Maximum date is not respected',
                invalidDateMessage: 'Date is invalid',
              }}
            />
          </>
        );
      }
    }
    return <BasicDateInputExample />;
