ScenarioManagerTreeList

The aim of this component is to easily display a scenario tree.

Props

datasets
array
required
scenarios
array
required
deleteScenario
func
required
moveScenario
func
required
userId
string
required
buildSearchInfo
func
required
buildDatasetInfo
func
required
labels
object
{ status: 'Run status', successful: 'Successful', failed: 'Failed', created: 'Created', dataset: 'Dataset', searchField: 'Search' }

Basic Usage

<div style={{ height: '500px' }}>
  <ScenarioManagerTreeList
    datasets={[
      {
        id: 'D-000001',
        name: 'A wonderful dataset name',
        tags: ['dataset'],
      },
      {
        id: 'D-000002',
        name: 'Second wonderful dataset name',
        tags: ['dataset'],
      },
    ]}
    scenarios={[
      {
        id: 'S-00001',
        name: 'Value 1',
        parentId: null,
        ownerId: '123456',
        creationDate: '2021-09-22T15:53:22.622797996Z',
        datasetList: ['D-000001'],
        ownerName: 'Bob Bobinson',
        state: 'Running',
      },
      {
        id: 'S-00002',
        name: 'Value 2',
        parentId: 'S-00001',
        ownerId: '000000',
        creationDate: '2021-09-22T16:53:22.622797996Z',
        datasetList: ['D-000001'],
        ownerName: 'Bob Bobinson',
        state: 'Created',
      },
      {
        id: 'S-00003',
        name: 'Value 3',
        parentId: null,
        ownerId: '000000',
        creationDate: '2021-09-22T17:53:22.622797996Z',
        datasetList: ['D-000001'],
        ownerName: 'Bob Bobinson',
        state: 'Failed',
      },
    ]}
    currentScenarioId={'S-00001'}
    userId={'123456'}
    deleteScenario={scenarioId => console.log(scenarioId)}
    moveScenario={moveData => console.log(moveData)}
    buildSearchInfo={scenarioCount => {
      return scenarioCount + ' Scenario found'
    }}
    buildDatasetInfo={() => {
      return 'Dataset'
    }}
  />
</div>
